String toHexString(const unsigned char* data, int length) {
  String hexStr="";
  for(int i=0;i<length;i++){
    if(data[i]<16) hexStr+="0";
    hexStr+=String(data[i],HEX);
  }
  hexStr.toUpperCase();
  return hexStr;
}

// ---- SAFE BUFFERS ----
uint8_t rawCC3[64];
uint8_t rawHR[4];
int rawCC3Len=0;
int rawHRLen=0;

void onCustomChar3Notify(BLEDevice, BLECharacteristic c){
  rawCC3Len = c.valueLength();
  if(rawCC3Len > 64) rawCC3Len=64;
  memcpy(rawCC3, c.value(), rawCC3Len);
  lastServiceReceived = millis();
}

void onHeartRateNotify(BLEDevice, BLECharacteristic c){
  rawHRLen = c.valueLength();
  if(rawHRLen > 4) rawHRLen=4;
  memcpy(rawHR, c.value(), rawHRLen);
  lastServiceReceived = millis();
}

void setECGStreaming(BLEDevice p){

  BLECharacteristic cmd=
    p.characteristic("9f154f02-2020-11e6-8749-0002a5d5c51b");

  if(!cmd) return;

  uint8_t c[2]={0x30,0x80};
  cmd.writeValue(c,2,false);
}

// -----------------------------------------------------

bool setupPeripheral(BLEDevice p){

  if(!p.connect()) return false;
  if(!p.discoverAttributes()) return false;

  BLECharacteristic c3=
    p.characteristic("9f154f03-2020-11e6-8749-0002a5d5c51b");

  if(c3){
    c3.subscribe();
    c3.setEventHandler(BLECharacteristicEvent::BLEUpdated,
                       onCustomChar3Notify);
  }

  BLECharacteristic hr=
    p.characteristic("2a37");

  if(hr){
    hr.subscribe();
    hr.setEventHandler(BLECharacteristicEvent::BLEUpdated,
                       onHeartRateNotify);
  }

  setECGStreaming(p);

  lastServiceReceived=millis();
  return true;
}

String sendAT(const String& cmd, int timeout=1000) {

  xSemaphoreTake(modemMutex, portMAX_DELAY);

  while(modemSerial.available())
    modemSerial.read();

  modemSerial.println(cmd);

  unsigned long t = millis();
  String r;

  while(millis() - t < (unsigned)timeout) {

    while(modemSerial.available()) {

      char c = modemSerial.read();
      r += c;
    }

    vTaskDelay(1);
  }

  xSemaphoreGive(modemMutex);

  return r;
}

String getValue(const String &data, char sep, int index) {
  int found=0,start=0;
  for(int i=0;i<data.length();i++){
    if(data[i]==sep||i==data.length()-1){
      if(found==index)
        return data.substring(start,i==data.length()-1?i+1:i);
      found++; start=i+1;
    }
  }
  return "";
}

void parseGNSS(const String& r) {

  int p=r.indexOf("+QGPSLOC:");
  if(p<0) return;

  p=r.indexOf(':',p);
  if(p<0) return;

  String d=r.substring(p+1);
  d.trim();

  String lat=getValue(d,',',1);
  String lon=getValue(d,',',2);
  String acc=getValue(d,',',3);

  if(lat.length()&&lon.length()) {

    xSemaphoreTake(dataMutex,portMAX_DELAY);

    data.Lat=lat.toDouble();
    data.Lon=lon.toDouble();
    data.Acc=acc.toFloat();
    data.gnssValid= true;
    data.lastGnssUpdate = millis(); 

    xSemaphoreGive(dataMutex);
  }
}

bool modemNetworkReady(){

  String r;

  r = sendAT("AT+CGPADDR=1", 500);

  if(r.indexOf(".") >= 0)
    return true;

  return false;
}

void mqttInit(){
  LOG("[MQTT] Initializing session");

  // cleanup stale sockets only if needed
  sendAT("AT+QMTCLOSE=0",500);

  // ---- WAIT FOR NETWORK ----
  for(int i=0;i<3;i++){
    if(modemNetworkReady()) break;
    LOG("[MQTT] waiting for network...");
    delay(1000);
  }

  //sendAT("AT+QMTDISC=0", 2000);
  sendAT("AT+QIACT=1",5000);

  String r;

  sendAT("AT+QMTCFG=\"keepalive\",0,120",1000);

  r = sendAT(
    String("AT+QMTOPEN=0,\"") +
    mqttServer + "\"," + mqttPort,
    5000
  );

  if(r.indexOf("+QMTOPEN: 0,0") == -1){

    mqttConnected = false;
    LOG("[MQTT] open failed");
    return;
  }

  r = sendAT(
    String("AT+QMTCONN=0,\"") +
    mqttClientId +
    "\",\"" +
    mqttUser +
    "\",\"" +
    mqttPassword +
    "\"",
    6000
  );

  if(r.indexOf("+QMTCONN: 0,0,0") != -1){
    mqttConnected = true;
    lastMqttOk = millis();
    LOG("[MQTT] Connected");
  }else{
    mqttConnected = false;
    LOG("[MQTT] Connect failed");
  }
}

void mqttPublish(const String& payload){

  if(!mqttConnected){

    if(millis() - lastReconnectAttempt > 3000){

      lastReconnectAttempt = millis();
      mqttInit();
    }

    if(!mqttConnected){
      LOG("[MQTT] reconnect failed");
      return;
    }
  }

  xSemaphoreTake(modemMutex, portMAX_DELAY);

  while(modemSerial.available())
    modemSerial.read();

  modemSerial.print("AT+QMTPUB=0,0,0,0,\"");
  modemSerial.print(mqttTopic);
  modemSerial.println("\"");

  delay(100);

  modemSerial.print(payload);
  modemSerial.write(0x1A);

  String r;
  unsigned long t = millis();

  while(millis() - t < 7000){

    while(modemSerial.available()){

      char c = modemSerial.read();
      r += c;

      // REAL SUCCESS URC
      if(r.indexOf("+QMTPUB: 0,0,0") != -1){

        mqttConnected = true;

        xSemaphoreGive(modemMutex);

        LOG("[MQTT] " + payload);
        return;
      }

      // modem/broker error
      if(r.indexOf("ERROR") != -1){

        mqttConnected = false;

        xSemaphoreGive(modemMutex);

        LOG("[MQTT] publish failed");
        return;
      }
    }

    vTaskDelay(1);
  }

  xSemaphoreGive(modemMutex);

  // IMPORTANT:
  // timeout != disconnect
  LOG("[MQTT] publish timeout");
}

// PUBLISH TASK
// ===============================================================
void publishTask(void*){

  StaticJsonDocument<512> doc;

  mqttInit();
  vTaskDelay(2000/portTICK_PERIOD_MS);

  for(;;){

    if(millis()-lastSendTime >= sendInterval*1000UL){

      xSemaphoreTake(dataMutex,portMAX_DELAY);

      doc.clear();
      doc["id"]=data.id;

      if(data.bleValid){
        doc["B"]=data.B;
        doc["HR"]=data.HR;
        doc["CC1"]=data.CC1;
        doc["CC2"]=data.CC2;
        doc["CC3"]=data.CC3;
      }

      // ---- GNSS MERGE ONLY IF FRESH ----
      if(data.gnssValid) {
        doc["Lat"] = data.Lat;
        doc["Lon"] = data.Lon;
        doc["Acc"] = data.Acc;

        // immediately consume the fix
        data.gnssValid = false;
      }

      xSemaphoreGive(dataMutex);

      String out;
      serializeJson(doc,out);
      mesh.println(out);
      LOG("[MESH] " + out);

      mqttPublish(out);

      //LOG("[MQTT] "+out);

      lastSendTime=millis();
    }

    vTaskDelay(200/portTICK_PERIOD_MS);
  }
}

// BLE TASK  (Your logic preserved)
// ===============================================================
void vbleTask(void*){

  // ---------- INIT ----------
  if (!BLE.begin()) {
    LOG("[BLE] Start failed!");
    vTaskDelete(NULL);
  }

  LOG("[BLE] Central ready");

  unsigned long lastScanAttempt = 0;
  bool wasConnected = false;

  for(;;){

    BLE.poll();

    // =====================================================
    // 1) HANDLE DISCONNECT
    // =====================================================
    if (connectedPeripheral.connected() == false && wasConnected) {
      LOG("[BLE] Disconnected from device");

      connectedPeripheral = BLEDevice();

      autoReconnect = true;
      wasConnected = false;
    }

    // =====================================================
    // 2) RECONNECT STATE MACHINE (NON BLOCKING)
    // =====================================================
    if (!connectedPeripheral && autoReconnect) {

      // try every 8 seconds
      if (millis() - lastScanAttempt > 8000) {

        lastScanAttempt = millis();

        LOG("[BLE] Scanning for: " + savedDeviceAddr);

        BLE.scan();

        unsigned long s = millis();
        while (millis() - s < 4000) {

          BLEDevice d = BLE.available();

          if (d && String(d.address().c_str()) == savedDeviceAddr) {

            BLE.stopScan();

            LOG("[BLE] Found saved device");

            if (setupPeripheral(d)) {
              connectedPeripheral = d;
              autoReconnect = false;
              wasConnected = true;
              LOG("[BLE] Connected & setup OK");
            } else {
              LOG("[BLE] Setup failed");
            }
            break;
          }
          vTaskDelay(20);
        }

        BLE.stopScan();
      }
    }

    // =====================================================
    // 3) PROCESS NOTIFY BUFFERS (SAFE ZONE)
    // =====================================================
    if (rawCC3Len > 0) {
      xSemaphoreTake(dataMutex, portMAX_DELAY);

      data.CC3 = "0x" + toHexString(rawCC3, rawCC3Len);
      data.bleValid = true;

      rawCC3Len = 0;

      xSemaphoreGive(dataMutex);
    }

    if (rawHRLen >= 2) {

      uint8_t flags = rawHR[0];
      uint16_t hr = (flags & 0x01 && rawHRLen >= 3) ?
                     (rawHR[1] | (rawHR[2] << 8)) : rawHR[1];

      xSemaphoreTake(dataMutex, portMAX_DELAY);

      data.HR = hr;
      data.bleValid = true;

      rawHRLen = 0;

      xSemaphoreGive(dataMutex);
    }

    // =====================================================
    // 4) WATCHDOG ONLY WHEN EVER CONNECTED
    // =====================================================
    if (wasConnected &&
        millis() - lastServiceReceived > 12000) {

      LOG("[BLE] Watchdog timeout - reconnecting");

      if (connectedPeripheral.connected()) {
        connectedPeripheral.disconnect();
      }

      connectedPeripheral = BLEDevice();
      autoReconnect = true;
      wasConnected = false;
    }

    // =====================================================
    // 5) POLL STATIC CHARACTERISTICS
    // =====================================================
    if (connectedPeripheral &&
        connectedPeripheral.connected()) {

      BLECharacteristic c1 = connectedPeripheral.characteristic("9f154f01-2020-11e6-8749-0002a5d5c51b");
      BLECharacteristic batt = connectedPeripheral.characteristic("2a19");

      xSemaphoreTake(dataMutex, portMAX_DELAY);

      if (c1 && c1.canRead() && c1.read()) {
        data.CC1 = "0x" +
          toHexString(c1.value(), c1.valueLength());
      }

      if (batt && batt.canRead() && batt.read()) {
        data.B = batt.value()[0];
      }

      data.CC2 = "N/A";

      xSemaphoreGive(dataMutex);
    }

    vTaskDelay(50);
  }
}

// ==== Scan: return only devices whose name starts with "Frontier" ====
void handleScan() {
  BLE.scan();
  delay(3000);
  String json = "[";
  while (BLE.available()) {
    BLEDevice d = BLE.available();
    String nm = String(d.localName().c_str());
    if (nm.length() && nm.startsWith("Frontier")) {
      json += "{\"name\":\"" + nm +
              "\",\"addr\":\"" + String(d.address().c_str()) + "\"},";
    }
  }
  if (json.endsWith(",")) json.remove(json.length()-1);
  json += "]";
  BLE.stopScan();
  server.send(200, "application/json", json);
}

void handleSelect() {
  if (!server.hasArg("addr")) {
    server.send(400, "text/plain", "Missing addr"); return;
  }
  String addr = server.arg("addr");
  String name = "";
  if (server.hasArg("name")) name = server.arg("name");

  savedDeviceAddr = addr;
  savedDeviceName = name;
  preferences.putString("ble_addr", savedDeviceAddr);
  preferences.putString("ble_name", savedDeviceName);
  // Clear any existing nickname for previous device - keep only one saved device nickname for simplicity
  // (If you want per-device nicknames, change storage strategy)
  preferences.putString("ble_nick", preferences.getString("ble_nick", "")); // keep existing nick if any
  server.send(200, "text/plain", "Device saved: " + savedDeviceName + " (" + savedDeviceAddr + ")");
  autoReconnect = true;
  delay(500);
  ESP.restart();   // 🔄 restart right after selecting a device
}

void handleClear() {
  if (connectedPeripheral.connected()) connectedPeripheral.disconnect();
  preferences.remove("ble_addr");
  preferences.remove("ble_name");
  preferences.remove("ble_nick");
  savedDeviceAddr = "";
  savedDeviceName = "";
  savedDeviceNick = "";
  server.send(200, "text/plain", "Device cleared. Please scan again.");
}

void handleSetInterval() {
  if (!server.hasArg("sec")) {
    server.send(400, "text/plain", "Missing sec"); 
    return;
  }
  sendInterval = server.arg("sec").toInt();
  if (sendInterval < 1) sendInterval = 1; // minimum 1 second
  preferences.putUInt("send_interval", sendInterval);
  server.send(200, "text/plain", "Interval updated: " + String(sendInterval) + " seconds");
}

void handleSetNick() {
  if (!server.hasArg("addr") || !server.hasArg("nick")) {
    server.send(400, "text/plain", "Missing addr or nick"); return;
  }
  String addr = server.arg("addr");
  String nick = server.arg("nick");
  // Only allow setting nick for currently saved device addr
  if (addr != savedDeviceAddr) {
    server.send(400, "text/plain", "Address mismatch with saved device.");
    return;
  }
  savedDeviceNick = nick;
  preferences.putString("ble_nick", savedDeviceNick);
  server.send(200, "text/plain", "Nickname saved: " + savedDeviceNick);
}

void handleLog(){

  String page = R"rawliteral(
  <html>
  <head>
    <meta http-equiv="refresh" content="3">
    <style>
      body{background:#111;color:#0f0;font-family:monospace;}
      pre{white-space:pre-wrap;}
    </style>
  </head>
  <body>
  <h3>ESP32 LIVE LOG</h3>
  <pre>
  )rawliteral";

  // show newest last
  for(int i=0;i<LOG_LINES;i++){
    int idx = (logIndex + i) % LOG_LINES;
    if(logBuffer[idx].length())
      page += logBuffer[idx] + "\n";
  }

  page += "</pre></body></html>";

  server.send(200, "text/html", page);
}

void setup(){
  Serial.begin(115200);
  mesh.begin(115200, SERIAL_8N1, 41, 42);

  pinMode(MODEM_PWRKEY, OUTPUT);
  digitalWrite(MODEM_PWRKEY, LOW);
  delay(200);
  digitalWrite(MODEM_PWRKEY, HIGH);
  delay(1000);
  digitalWrite(MODEM_PWRKEY, LOW);

  modemSerial.begin(115200,SERIAL_8N1,RXD1,TXD1);
  mqttClientId = "ESP32_" + String((uint32_t)ESP.getEfuseMac(), HEX);

  delay(8000);

  WiFi.softAP(ssid,password);

  preferences.begin("ble",false);
  savedDeviceAddr = preferences.getString("ble_addr", "");
  savedDeviceName = preferences.getString("ble_name", "");
  savedDeviceNick = preferences.getString("ble_nick", "");
  sendInterval = preferences.getUInt("send_interval", 5); // default 5s

  // ---- WEB ROUTES (YOUR ORIGINAL) ----
  server.on("/", handleRoot);
  server.on("/scan", handleScan);
  server.on("/select", handleSelect);
  server.on("/clear", handleClear);
  server.on("/reboot", handleReboot);
  server.on("/setnick", handleSetNick);
  server.on("/setInterval", handleSetInterval);
  server.on("/log", handleLog);
  server.begin();

  if(savedDeviceAddr.length()> 0){
    autoReconnect=true;
  }

  dataMutex=xSemaphoreCreateMutex();
  modemMutex = xSemaphoreCreateMutex();

  xTaskCreatePinnedToCore(webTask,"web",8000,NULL,1,NULL,0);
  xTaskCreatePinnedToCore(vbleTask,"ble",10000,NULL,3,NULL,1);
  xTaskCreatePinnedToCore(gnssTask,"gnss",8000,NULL,1,NULL,1);
  xTaskCreatePinnedToCore(publishTask,"pub",9000,NULL,1,NULL,1);
}
