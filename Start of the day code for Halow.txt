#define LED_POWER   GPIO_NUM_1
#define LED_MQTT    GPIO_NUM_2

#define UART_PORT      UART_NUM_1
#define UART_RX_PIN    44
#define UART_BUF_SIZE  8192

static bool mqtt_connected = false;

static esp_mqtt_client_handle_t mqtt_client;
static volatile int64_t mqtt_led_off_time = 0;

static const char *TAG = "HALOW_MQTT";

void led_task(void *pvParameters)
{
    while (1)
    {
        if (mqtt_led_off_time != 0)
        {
            if (esp_timer_get_time() > mqtt_led_off_time)
            {
                gpio_set_level(LED_MQTT, 0);
                mqtt_led_off_time = 0;
            }
        }

        vTaskDelay(pdMS_TO_TICKS(5));
    }
}

void uart_init(void)
{
    uart_config_t uart_config = {
        .baud_rate = 115200,
        .data_bits = UART_DATA_8_BITS,
        .parity    = UART_PARITY_DISABLE,
        .stop_bits = UART_STOP_BITS_1,
        .flow_ctrl = UART_HW_FLOWCTRL_DISABLE,
        .source_clk = UART_SCLK_DEFAULT,
    };

    uart_driver_install(
        UART_PORT,
        UART_BUF_SIZE,
        0,
        0,
        NULL,
        0);

    uart_param_config(UART_PORT, &uart_config);

    uart_set_pin(
      UART_PORT,
      UART_PIN_NO_CHANGE,
      UART_RX_PIN,
      UART_PIN_NO_CHANGE,
      UART_PIN_NO_CHANGE);

    ESP_LOGI(TAG, "UART Initialized");
}

static QueueHandle_t mqtt_queue;

typedef struct
{
    char data[512];
} mqtt_msg_t;

void uart_task(void *pvParameters)
{
    uint8_t rx_buffer[512];

    static char line_buffer[512];
    int line_index = 0;

    while (1)
    {
        int len = uart_read_bytes(
            UART_PORT,
            rx_buffer,
            sizeof(rx_buffer),
            pdMS_TO_TICKS(20));

        if (len > 0)
        {
            for (int i = 0; i < len; i++)
            {
                char ch = rx_buffer[i];

                if (ch == '\n')
                {
                    line_buffer[line_index] = '\0';

                    if (line_index > 0)
                    {
                        mqtt_msg_t msg;

                        strncpy(msg.data,
                          line_buffer,
                          sizeof(msg.data) - 1);

                  msg.data[sizeof(msg.data) - 1] = '\0';

                        static uint32_t dropped = 0;

                        if (xQueueSend(mqtt_queue, &msg, 0) != pdPASS)
                        {
                            dropped++;

                            if ((dropped % 100) == 0)
                            {
                                ESP_LOGW(TAG, "Dropped %lu packets", dropped);
                            }
                        }
                    }

                    line_index = 0;
                }
                else
                {
                    if (line_index < sizeof(line_buffer) - 1)
                    {
                        line_buffer[line_index++] = ch;
                    }
                    else
                    {
                        // overflow protection
                        line_index = 0;
                    }
                }
            }
        }
    }
}

static void mqtt_event_handler(void *handler_args,
                               esp_event_base_t base,
                               int32_t event_id,
                               void *event_data)
{
    switch ((esp_mqtt_event_id_t)event_id)
    {       
        case MQTT_EVENT_CONNECTED:

          mqtt_connected = true;
          ESP_LOGI(TAG, "MQTT CONNECTED");

          break;

        case MQTT_EVENT_DISCONNECTED:

          mqtt_connected = false;
          ESP_LOGI(TAG, "MQTT DISCONNECTED");

          break;
        
        case MQTT_EVENT_ERROR:

          ESP_LOGE(TAG, "MQTT ERROR");

          break;

        default:
            break;
    }
}

void mqtt_publish_task(void *pvParameters)
{
    mqtt_msg_t msg;

    static char batch_buffer[8192];
    int batch_len = 0;

    int64_t last_publish = esp_timer_get_time();

    while (1)
    {
        if (xQueueReceive(mqtt_queue,
                          &msg,
                          pdMS_TO_TICKS(20)))
        {
            int msg_len = strlen(msg.data);

            if ((batch_len + msg_len + 2) < sizeof(batch_buffer))
            {
                memcpy(&batch_buffer[batch_len],
                       msg.data,
                       msg_len);

                batch_len += msg_len;

                batch_buffer[batch_len++] = '\n';
                batch_buffer[batch_len] = '\0';
            }
        }

        int64_t now = esp_timer_get_time();

        if (batch_len > 0 &&
            (now - last_publish) > 500000)
        {
            if (mqtt_connected)
            {
                  int msg_id = esp_mqtt_client_publish(
                  mqtt_client,
                  "v1/devices/me/telemetry",
                  batch_buffer,
                  batch_len,
                  0,
                  0);

              if (msg_id >= 0)
              {
                  gpio_set_level(LED_MQTT, 1);

                  mqtt_led_off_time =
                      esp_timer_get_time() + 20000;
              }
            }

            batch_len = 0;

            last_publish = now;
        }
    }
}

    esp_mqtt_client_config_t mqtt_cfg = {

void led_init(void)
{
    gpio_config_t io_conf = {
        .mode = GPIO_MODE_OUTPUT,
        .pin_bit_mask =
            (1ULL << LED_POWER) |
            (1ULL << LED_MQTT),
    };

    gpio_config(&io_conf);

    // Power LED ON
    gpio_set_level(LED_POWER, 1);

    // MQTT LED OFF initially
    gpio_set_level(LED_MQTT, 0);
}

                    else
                    {
                        // overflow reset
                        line_index = 0;
                    }
                }
            }
        }
    }
}

static void mqtt_event_handler(void *handler_args,
                               esp_event_base_t base,
                               int32_t event_id,
                               void *event_data)
{
    esp_mqtt_event_handle_t event = event_data;

    if (event->client == mqtt_client1)
    {
        switch ((esp_mqtt_event_id_t)event_id)
        {
            case MQTT_EVENT_CONNECTED:
                mqtt1_connected = true;
                ESP_LOGI(TAG, "MQTT1 CONNECTED");
                break;

            case MQTT_EVENT_DISCONNECTED:
                mqtt1_connected = false;
                ESP_LOGW(TAG, "MQTT1 DISCONNECTED");
                break;

            default:
                break;
        }
    }

    if (event->client == mqtt_client2)
    {
        switch ((esp_mqtt_event_id_t)event_id)
        {
            case MQTT_EVENT_CONNECTED:
                mqtt2_connected = true;
                ESP_LOGI(TAG, "MQTT2 CONNECTED");
                break;

            case MQTT_EVENT_DISCONNECTED:
                mqtt2_connected = false;
                ESP_LOGW(TAG, "MQTT2 DISCONNECTED");
                break;

            default:
                break;
        }
    }
}

void mqtt_publish_task(void *pvParameters)
{
    mqtt_msg_t msg;
    mqtt_msg_t temp;

    while (1)
    {
        // wait until MQTT reconnects
        if (!mqtt1_connected && !mqtt2_connected)
        {
            vTaskDelay(pdMS_TO_TICKS(500));
            continue;
        }

        // get next UART packet
        if (xQueueReceive(
                mqtt_queue,
                &msg,
                pdMS_TO_TICKS(100)))
        {
            // publish to MQTT
            if (mqtt1_connected)
            {
                int id = esp_mqtt_client_enqueue(
                  mqtt_client1,
                  "/Galore/asset_tracking/send",
                  msg.data,
                  0,
                  1,
                  0,
                  true);

                if (id < 0)
                {
                    ESP_LOGW(TAG, "MQTT1 publish failed");
                }
                vTaskDelay(pdMS_TO_TICKS(10));
            }

            if (mqtt2_connected)
            {
                int id = esp_mqtt_client_enqueue(
                  mqtt_client2,
                  "v1/devices/me/telemetry",
                  msg.data,
                  0,
                  0,
                  0,
                  true);

                if (id < 0)
                {
                    ESP_LOGW(TAG, "MQTT2 publish failed");
                }
                vTaskDelay(pdMS_TO_TICKS(10));
            }

            // IMPORTANT:
            // limit MQTT publish rate
            // prevents socket overload + ping timeout
            vTaskDelay(pdMS_TO_TICKS(5));
        }
    }
}

void app_main(void)
{
    printf("HaLow MQTT Gateway\n");

    app_wlan_init();
    app_wlan_start();

    ESP_LOGI(TAG, "Waiting for HaLow network...");

    vTaskDelay(pdMS_TO_TICKS(25000));

    // CREATE QUEUE FIRST
    mqtt_queue = xQueueCreate(150, sizeof(mqtt_msg_t));

    if (mqtt_queue == NULL)
    {
        ESP_LOGE(TAG, "Failed to create MQTT queue");
        return;
    }

    uart_init();

    mqtt_app_start();

    xTaskCreatePinnedToCore(
        uart_task,
        "uart_task",
        8192,
        NULL,
        10,
        NULL,
        1);

    xTaskCreatePinnedToCore(
        mqtt_publish_task,
        "mqtt_publish_task",
        8192,
        NULL,
        9,
        NULL,
        1);
}
